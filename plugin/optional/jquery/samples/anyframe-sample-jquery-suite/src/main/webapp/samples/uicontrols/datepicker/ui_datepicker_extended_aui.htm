  • useExtSelectBoolean Type

    브라우저의 기본 Select Box 대신, Customized 된 Select Box 를 사용하도록 지정한다. 기본값은 false. {true / false}
    
        
    

    // initialize
$(".selector").datepicker({useExtSelect: true});

// getter
var useExtSelect = $(".selector").datepicker("option", "useExtSelect");

// setter
$(".selector").datepicker("option", "useExtSelect", true);
