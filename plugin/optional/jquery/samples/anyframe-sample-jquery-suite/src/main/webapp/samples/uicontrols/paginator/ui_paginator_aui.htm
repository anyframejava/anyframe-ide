Options

        

            
    

                
  • itemsTotalCountNumber Type
    • 
                
    전체 데이타(레코드) 개수를 지정한다. itemsTotalCount 와 itemsPerPage 값에 의해 화면에 보여질 페이지 개수가 결정된다.
    
                
    

    // initialize
$(".selector").paginator({itemsTotalCount: 100});

// getter
var totalCount = $(".selector").paginator("option", "itemsTotalCount");

// setter
$(".selector").paginator("option", "itemsTotalCount", 100);

    
                
    

                
  • itemsPerPageNumber Type
    • 
                
    각 페이지에 표현될 데이타(레코드) 개수를 지정한다. 
    
                
    

    // initialize
$(".selector").paginator({itemsPerPage: 10});

// getter
var itemCount = $(".selector").paginator("option", "itemsPerPage");

// setter
$(".selector").paginator("option", "itemsPerPage", 10);

    
                
    

                
  • ellipsisTextString Type
    • 
                
    페이지 개수가 10개를 넘는 경우, 첫 페이지 또는 마지막 페이지를 구분 지어주는 문자. 일반적으로 "..." 또는 "~" 형태로 표현된다.
    
                
    

    // initialize
$(".selector").paginator({ellipsisText: 10});

// getter
var oldText = $(".selector").paginator("option", "ellipsisText");

// setter
$(".selector").paginator("option", "ellipsisText", "~");

    
                
    


                
  • isFirstPageZeroBoolean Type
    • 
                
    첫번째 페이지번호를 0 (Zero) 부터 시작하게 할 지를 결정한다. 기본값은 false.
    
                
    

    // initialize
$(".selector").paginator({isFirstPageZero: true});

// getter
var oldFlag = $(".selector").paginator("option", "isFirstPageZero");

// setter
$(".selector").paginator("option", "isFirstPageZero", true);

    
                
    


                
  • buttonTextObject Type
    • 
                
    아래와 같은 형태로 값을 설정하여, 각 4개의 버튼 레이블을 변경할 수 있다. 
    
                { first: "first레이블", prev: "prev레이블", next: "next레이블", last: "last레이블" }
                
    
                
    

    // initialize
$(".selector").paginator({buttonText: { first: "first", prev: "prev", next: "next", last: "last"}});

// getter
var oldLabelObject = $(".selector").paginator("option", "buttonText");

// setter
$(".selector").paginator("option", "buttonText", true, { first: "처음", prev: "이전", next: "다음", last: "마지막"});

    
                
    


                
  • buttonStyleModeNumber Type
    • 
                
    buttonText 로 설정되는 4개의 버튼은, 이미지 또는 텍스트 또는 이미지+텍스트 형태로 표현되며, 아래 숫자값 중 하나를 선택하여 결정된다. 기본값은 0.
    
                   0 : 텍스트 
    
                   1 : 이미지 
    
                   2 : 이미지 + 텍스트 
    
                
    
                
    

    // initialize
$(".selector").paginator({buttonStyleMode: 1});

// getter
var oldValue = $(".selector").paginator("option", "buttonStyleMode");

// setter
$(".selector").paginator("option", "buttonStyleMode", 2);

    
                
    

                
  • isAlwaysShowButtonBoolean Type
    • 
                
     first, prev, next,last 버튼을 항상 보이게 할 지를 결정한다. 내부적으로 페이지 버튼을 보여주는 개수에 대하여 제한을 두고 있으며, 
    
                    그 값을 초과하여 페이지가 구성될 때 자동으로 4개의 버튼이 보이게 되는데 isAlwaysShowButton 값이 true 인 경우에는 예외적으로 항상 보이도록 한다. 
    
                    기본값은 false.
    
                
    

    // initialize
$(".selector").paginator({isFirstPageZero: true});

// getter
var oldFlag = $(".selector").paginator("option", "isFirstPageZero");

// setter
$(".selector").paginator("option", "isFirstPageZero", true);

    
                
    

                
  • changeFunction Type
    • 
                
    function(pageNum), 페이지 변경이 일어날때 호출되는 함수이다. 기본값은 null.
    
                pageNum: 선택된 페이지 값. (Number Type)
                
    
                
    

    // code example
$(".selector").paginator({ change: function(pageNum} {...} });

    
                
    


        

        


        
Methods

        


            
    

                
  • getCurrentPage()return : Number Type
    • 
                
    선택된 페이지 번호를 반환한다.
    
        
    

    // code example
var curPageNum = $(".selector").paginator("getCurrentPage");

    
        
    

                
  • setCurrentPage( Number newPage )
    • 
                
    입력된 페이지를 선택하여 보여준다.
    
        
    

    // code example
$(".selector").paginator("setCurrentPage", 5);
